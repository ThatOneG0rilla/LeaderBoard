handlers.HandlePlayerReport = function (Args, Context) {
    var ModeratorIds = ["MODERATOR_ID"];
    var ReasonDurations = {
        "Hate Speech": 2592000,
        "Cheating": 1209600,
        "Toxicity": 604800
    };

    var IsModerator = ModeratorIds.indexOf(Args.ReporterId) !== -1;

    if (IsModerator) {
        var Duration = ReasonDurations[Args.Reason] || 4;
        server.BanUsers({
            PlayFabIds: [Args.TargetId],
            DurationInSeconds: Duration
        });
        return { Result: "Moderator Ban Executed" };
    }

    var ReportText =
        Args.TargetId + " Was Reported By " + Args.ReporterId + "\\n" +
        "Reporter Colour: " + Args.ReporterColor + "\\n" +
        "Reported Colour: " + Args.TargetColor + "\\n" +
        "Room: " + Args.Room + "\\n" +
        "Reporter Username: " + Args.ReporterName + "\\n" +
        "Reported Username: " + Args.TargetName;

    var Payload = JSON.stringify({ content: ReportText });
    var WebhookUrl = "YOUR_DISCORD_WEBHOOK_URL";

    try {
        http.request(WebhookUrl, "post", {
            headers: { "Content-Type": "application/json" },
            body: Payload
        });
    } catch (Error) {
        log.error("Webhook Failed: " + Error.message);
    }
    return { Result: "Report Sent To Discord" };
};
